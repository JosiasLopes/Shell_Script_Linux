#!/bin/bash

echo "Criando diretorios"

sudo mkdir /publico
sudo chown root /publico
sudo mkdir /admin
sudo chown root /admin
sudo chmod 770
sudo mkdir /ven
sudo chown root /ven
sudo chmod 770
sudo mkdir /sec
sudo chown root /sec
sudo chmod 770

echo "Criando Grupos"
sudo addgroup grp_admin
sudo addgroup grp_ven
sudo addgroup grp_sec

echo "Liberando acesso total a pasta publica"
sudo chmod /publico 777

echo "Adicionando usuario ao Grupo de Adminisradores"
sudo useradd -g grp_admin Carlos -c "Um dos administradores" -s /bin/bash -m
sudo useradd -g grp_admin Maria -c "Um dos administradores" -s /bin/bash -m 
sudo useradd -g grp_admin Joao_ -c "Um dos administradores" -s /bin/bash -m 

echo "Adicionando usuarios ao Grupo GRP_VEN"
sudo useradd -g grp_ven Debora -s /bin/bash -m 
sudo useradd -g grp_ven Sebastiana -s /bin/bash -m 
sudo useradd -g grp_ven Roberto -s /bin/bash -m 


echo "Adicionando usuarios ao Grupo GRP_SEC"
sudo useradd -g grp_sec Josefina -s /bin/bash -m 
sudo useradd -g grp_sec Amanda -s /bin/bash -m 
sudo useradd -g grp_sec Rogerio -s /bin/bash -m
